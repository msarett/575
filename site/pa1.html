   /**
    *  Fill the rectangle with the color into the bitmap (already allocated).
    *
    *  The affected pixels are those whose centers are "contained" inside the rectangle:
    *      e.g. contained == center > min_edge && center <= max_edge
    *
    *  Any area in the rectangle that is outside of the bounds of the bitmap is ignored.
    *
    *  If the color's alpha is < 1, blend it with the previous pixel values using SRC_OVER blend mode.
    */
    void cs575_draw_rect(const GBitmap&, const GRect&, const GColor&)
